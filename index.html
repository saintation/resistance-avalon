import { test, expect, BrowserContext, Page } from '@playwright/test'; const TARGET_URL = 'https://saintation.github.io/resistance-avalon/'; const TOTAL_PLAYERS = 10; test.describe.serial('레지스탕스 아발론 10인 E2E 테스트 (풀 사이클)', () => { let contexts: BrowserContext[] = []; let pages: Page[] = []; let roomCode = ''; test.beforeAll(async ({ browser }) => { for (let i = 0; i < TOTAL_PLAYERS; i++) { const context = await browser.newContext(); const page = await context.newPage(); contexts.push(context); pages.push(page); } }); test.afterAll(async () => { for (const context of contexts) { await context.close(); } }); test('방장이 방을 생성하고 방 코드를 확인한다', async () => { const hostPage = pages[0]; await hostPage.goto(TARGET_URL); await hostPage.fill('#nicknameInput', '방장'); await hostPage.click('#createRoomBtn'); await expect(hostPage.locator('#roomTitle')).toHaveText(/방 코드: \d{4}/); const titleText = await hostPage.locator('#roomTitle').innerText(); roomCode = titleText.replace('방 코드: ', '').trim(); }); test('9명의 일반 플레이어가 방 코드를 통해 입장한다', async () => { test.setTimeout(90000); for (let i = 1; i < TOTAL_PLAYERS; i++) { const guestPage = pages[i]; await guestPage.goto(TARGET_URL); await guestPage.fill('#nicknameInput', `플레이어${i}`); await guestPage.fill('#roomCodeInput', roomCode); await guestPage.click('#joinRoomBtn'); await expect(guestPage.locator('#myNickname')).toHaveText(`플레이어${i}`); await expect(guestPage.locator('#playerList .player-item').filter({ hasText: `플레이어${i}` })).toBeVisible(); } const hostPage = pages[0]; await expect(hostPage.locator('#playerList .player-item')).toHaveCount(TOTAL_PLAYERS); }); test('모든 플레이어가 준비를 완료하면 방장이 10인 게임을 시작한다', async () => { test.setTimeout(90000); await Promise.all(pages.map(page => page.click('#readyBtn', { force: true }))); await Promise.all(pages.map(page => expect(page.locator('#readyBtn')).toHaveText('준비 완료!'))); const hostPage = pages[0]; await hostPage.check('#cbPercival', { force: true }); await hostPage.check('#cbMorgana', { force: true }); await hostPage.check('#cbMordred', { force: true }); await hostPage.check('#cbOberon', { force: true }); const startGameBtn = hostPage.locator('#startGameBtn'); await expect(startGameBtn).not.toBeDisabled(); await startGameBtn.click({ force: true }); for (let i = 0; i < TOTAL_PLAYERS; i++) { const page = pages[i]; await expect(page.locator('#playArea')).not.toHaveClass(/hidden/); await expect(page.locator('#gameActionArea')).toBeVisible(); await expect(page.locator('#myRoleDisplay')).not.toHaveText('로딩 중'); } }); test('모든 플레이어가 자신의 정체를 확인하고 지정된 롤 구성과 일치하는지 검증한다', async () => { const actualRoleCounts: Record = {}; for (let i = 0; i < TOTAL_PLAYERS; i++) { const page = pages[i]; await page.click('#toggleRoleBtn', { force: true }); await expect(page.locator('#roleSummaryContent')).toHaveClass(/show/); const roleText = await page.locator('#myRoleDisplay').innerText(); expect(roleText.length).toBeGreaterThan(0); expect(roleText).not.toBe('로딩 중'); actualRoleCounts[roleText] = (actualRoleCounts[roleText] || 0) + 1; await page.click('#toggleRoleBtn', { force: true }); await expect(page.locator('#roleSummaryContent')).not.toHaveClass(/show/); } expect(actualRoleCounts['멀린 (아서왕의 수하)']).toBe(1); expect(actualRoleCounts['암살자 (모드레드의 수하)']).toBe(1); expect(actualRoleCounts['퍼시벌 (아서왕의 수하)']).toBe(1); expect(actualRoleCounts['모르가나 (모드레드의 수하)']).toBe(1); expect(actualRoleCounts['모드레드 (모드레드의 수하)']).toBe(1); expect(actualRoleCounts['오베론 (모드레드의 수하)']).toBe(1); expect(actualRoleCounts['아서왕의 충직한 신하 (아서왕의 수하)']).toBe(4); }); test('원정 3회를 연속으로 성공시키고, 암살자 단계를 거쳐 게임이 종료된 후 대기실로 복귀한다', async () => { test.setTimeout(240000); for (let round = 1; round <= 3; round++) { let leaderPage: Page | null = null; await Promise.any( pages.map(page => page.locator('#teamSelectionArea').waitFor({ state: 'visible', timeout: 20000 }).then(() => page) ) ).then(page => { leaderPage = page; }).catch(() => {}); expect(leaderPage).not.toBeNull(); if (leaderPage) { // UI가 렌더링될 시간을 잠시 대기 await leaderPage.waitForTimeout(500); const requiredSizeText = await leaderPage.locator('#requiredTeamSize').innerText(); const requiredSize = parseInt(requiredSizeText, 10); const checkboxes = leaderPage.locator('.team-cb'); for (let i = 0; i < requiredSize; i++) { // force 옵션을 통해 스크롤 무시하고 강제 체크 진행 await checkboxes.nth(i).check({ force: true }); } await expect(leaderPage.locator('#submitTeamBtn')).not.toBeDisabled(); await leaderPage.locator('#submitTeamBtn').click({ force: true }); } await Promise.all(pages.map(page => page.locator('#teamVotingArea').waitFor({ state: 'visible', timeout: 10000 }) )); await Promise.all(pages.map(page => page.locator('#btnApprove').click({ force: true }) )); await Promise.all(pages.map(page => page.locator('#voteResultArea').waitFor({ state: 'visible', timeout: 15000 }) )); await Promise.all(pages.map(page => expect(page.locator('#gameStatusText')).toContainText('임무', { timeout: 15000 }) )); let questMemberCount = 0; let expectedRequiredSize = round === 1 ? 3 : 4; for (const page of pages) { if (await page.locator('#questVotingArea').isVisible()) { questMemberCount++; await page.locator('#btnQuestSuccess').click({ force: true }); } } expect(questMemberCount).toBe(expectedRequiredSize); await Promise.all(pages.map(page => page.locator('#questResultArea').waitFor({ state: 'visible', timeout: 15000 }) )); } await Promise.all(pages.map(page => page.locator('#assassinPhaseArea').waitFor({ state: 'visible', timeout: 15000 }) )); let assassinPage: Page | null = null; for (const page of pages) { if (await page.locator('#assassinActionUI').isVisible()) { assassinPage = page; break; } } expect(assassinPage).not.toBeNull(); if (assassinPage) { assassinPage.on('dialog', async dialog => { await dialog.accept(); }); await assassinPage.locator('#assassinTargetSelect').selectOption({ index: 1 }); await assassinPage.locator('#btnAssassinate').click({ force: true }); } await Promise.all(pages.map(page => page.locator('#gameOverArea').waitFor({ state: 'visible', timeout: 10000 }) )); const hostPage = pages[0]; await expect(hostPage.locator('#finalRoleList li')).toHaveCount(TOTAL_PLAYERS); await hostPage.locator('#btnRestartGame').click({ force: true }); await Promise.all(pages.map(page => page.locator('#waitingArea').waitFor({ state: 'visible', timeout: 10000 }) )); await Promise.all(pages.map(page => expect(page.locator('#playArea')).toHaveClass(/hidden/) )); await expect(hostPage.locator('#readyBtn')).toHaveText('준비하기'); }); });
